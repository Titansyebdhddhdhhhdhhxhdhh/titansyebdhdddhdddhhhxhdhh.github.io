============================================================================= # BACKTESTING FUNCTIONS # ============================================================================= def kupiec_pof(r, var_val, alpha=0.01): n = len(r) exc = int(np.sum(r < var_val)) if exc == 0 or exc == n: return exc, exc / n, 0.0, 1.0 p = exc / n LR = -2.0 * np.log( ((1 - alpha)**(n - exc) * alpha**exc) / ((1 - p)**(n - exc) * p**exc) ) return exc, p, float(LR), float(chi2.sf(LR, df=1)) def kupiec_tuff(r, var_val, alpha=0.01): idx = np.where(r < var_val)[0] if len(idx) == 0: return np.nan, np.nan v = int(idx[0]) + 1 logL0 = np.log(alpha + 1e-300) + (v - 1) * np.log(max(1 - alpha, 1e-300)) if v == 1: logL1 = 0.0 else: logL1 = np.log(1.0 / v) + (v - 1) * np.log(max(1.0 - 1.0/v, 1e-300)) LR = max(-2.0 * (logL0 - logL1), 0.0) return float(LR), float(chi2.sf(LR, df=1)) def mrb_basel(exc, T): exc_250 = int(round(exc / T * 250)) if exc_250 <= 4: return exc_250, "Verte", alpha=0.01): exc, p_obs, LR_pof, p_pof = kupiec_pof(r, var_val, alpha) hits = (r < var_val).astype(int) n00 = n01 = n10 = n11 = 0 for i in range(1, len(hits)): a, b = hits[i-1], hits[i] if a == 0 and
